#Example 1: Rename the local computer
Rename-Computer -NewName "Server044" -DomainCredential Domain01\Admin01 -Restart
Rename-Computer -NewName "Win10_local"





#Example 2: Rename a remote computer

#This command renames the Srv01 computer to Server001. The computer is not restarted.
#DomainCredential parameter specifies the credentials of a user who has permission to rename computers in the domain.
#Force parameter suppresses the confirmation prompt.

Rename-Computer -ComputerName "Srv01" -NewName "Server001" -DomainCredential Domain01\Admin01 -Force
