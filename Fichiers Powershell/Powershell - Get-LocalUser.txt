#Gets local user accounts. 
#This cmdlet gets default built-in user accounts, local user accounts that you created, and local accounts that you connected to Microsoft accounts.
Get-LocalUser

#Affiche les utilisateur locaux et toutes les colonnes infos
Get-LocalUser | Select *

Get-LocalUser | Select -ExcludeProperty PrincipalSource, ObjectClass | Export-csv -Path "$env:TEMP\Get-LocalUsers_$(get-date -f dd-MM-yyyy).csv" -NoTypeInformation





Get-LocalGroup   

#Get local Groups with attributes
Get-LocalGroup | Select Name, Description, SID | Export-csv -Path "$env:TEMP\Get-LocalGroups_$(get-date -f dd-MM-yyyy).csv" -NoTypeInformation
